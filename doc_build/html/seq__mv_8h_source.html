190 /*--------------------------------------------------------------------------
191  * hypre_Vector
192  *--------------------------------------------------------------------------*/
193 
194 typedef struct
195 {
196  HYPRE_Complex *data;
197  HYPRE_Int size;
198 
199  /* Does the Vector create/destroy `data'? */
200  HYPRE_Int owns_data;
201 
202  /* For multivectors...*/
203  HYPRE_Int num_vectors; /* the above "size" is size of one vector */
204  HYPRE_Int multivec_storage_method;
205  /* ...if 0, store colwise v0[0], v0[1], ..., v1[0], v1[1], ... v2[0]... */
206  /* ...if 1, store rowwise v0[0], v1[0], ..., v0[1], v1[1], ... */
207  /* With colwise storage, vj[i] = data[ j*size + i]
208  With rowwise storage, vj[i] = data[ j + num_vectors*i] */
209  HYPRE_Int vecstride, idxstride;
210  /* ... so vj[i] = data[ j*vecstride + i*idxstride ] regardless of row_storage.*/
211 
212 } hypre_Vector;
