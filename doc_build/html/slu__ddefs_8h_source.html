37 /*
38  * Global data structures used in LU factorization -
39  *
40  * nsuper: #supernodes = nsuper + 1, numbered [0, nsuper].
41  * (xsup,supno): supno[i] is the supernode no to which i belongs;
42  * xsup(s) points to the beginning of the s-th supernode.
43  * e.g. supno 0 1 2 2 3 3 3 4 4 4 4 4 (n=12)
44  * xsup 0 1 2 4 7 12
45  * Note: dfs will be performed on supernode rep. relative to the new
46  * row pivoting ordering
47  *
48  * (xlsub,lsub): lsub[*] contains the compressed subscript of
49  * rectangular supernodes; xlsub[j] points to the starting
50  * location of the j-th column in lsub[*]. Note that xlsub
51  * is indexed by column.
52  * Storage: original row subscripts
53  *
54  * During the course of sparse LU factorization, we also use
55  * (xlsub,lsub) for the purpose of symmetric pruning. For each
56  * supernode {s,s+1,...,t=s+r} with first column s and last
57  * column t, the subscript set
58  * lsub[j], j=xlsub[s], .., xlsub[s+1]-1
59  * is the structure of column s (i.e. structure of this supernode).
60  * It is used for the storage of numerical values.
61  * Furthermore,
62  * lsub[j], j=xlsub[t], .., xlsub[t+1]-1
63  * is the structure of the last column t of this supernode.
64  * It is for the purpose of symmetric pruning. Therefore, the
65  * structural subscripts can be rearranged without making physical
66  * interchanges among the numerical values.
67  *
68  * However, if the supernode has only one column, then we
69  * only keep one set of subscripts. For any subscript interchange
70  * performed, similar interchange must be done on the numerical
71  * values.
72  *
73  * The last column structures (for pruning) will be removed
74  * after the numercial LU factorization phase.
75  *
76  * (xlusup,lusup): lusup[*] contains the numerical values of the
77  * rectangular supernodes; xlusup[j] points to the starting
78  * location of the j-th column in storage vector lusup[*]
79  * Note: xlusup is indexed by column.
80  * Each rectangular supernode is stored by column-major
81  * scheme, consistent with Fortran 2-dim array storage.
82  *
83  * (xusub,ucol,usub): ucol[*] stores the numerical values of
84  * U-columns outside the rectangular supernodes. The row
85  * subscript of nonzero ucol[k] is stored in usub[k].
86  * xusub[i] points to the starting location of column i in ucol.
87  * Storage: new row subscripts; that is subscripts of PA.
88  */
89 typedef struct {
90  int *xsup; /* supernode and column mapping */
91  int *supno;
92  int *lsub; /* compressed L subscripts */
93  int *xlsub;
94  double *lusup; /* L supernodes */
95  int *xlusup;
96  double *ucol; /* U columns */
97  int *usub;
98  int *xusub;
99  int nzlmax; /* current max size of lsub */
100  int nzumax; /* " " " ucol */
101  int nzlumax; /* " " " lusup */
102  int n; /* number of columns in the matrix */
103  LU_space_t MemModel; /* 0 - system malloc'd; 1 - user provided */
104 } GlobalLU_t;
