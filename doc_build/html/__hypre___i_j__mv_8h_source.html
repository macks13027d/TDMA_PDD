43 typedef struct
44 {
45  HYPRE_Int local_num_rows; /* defines number of rows on this processors */
46  HYPRE_Int local_num_cols; /* defines number of cols of diag */
47 
48  HYPRE_Int need_aux; /* if need_aux = 1, aux_j, aux_data are used to
49  generate the parcsr matrix (default),
50  for need_aux = 0, data is put directly into
51  parcsr structure (requires the knowledge of
52  offd_i and diag_i ) */
53 
54  HYPRE_Int *row_length; /* row_length_diag[i] contains number of stored
55  elements in i-th row */
56  HYPRE_Int *row_space; /* row_space_diag[i] contains space allocated to
57  i-th row */
58  HYPRE_Int **aux_j; /* contains collected column indices */
59  HYPRE_Complex **aux_data; /* contains collected data */
60 
61  HYPRE_Int *indx_diag; /* indx_diag[i] points to first empty space of portion
62  in diag_j , diag_data assigned to row i */
63  HYPRE_Int *indx_offd; /* indx_offd[i] points to first empty space of portion
64  in offd_j , offd_data assigned to row i */
65  HYPRE_Int max_off_proc_elmts; /* length of off processor stash set for
66  SetValues and AddTOValues */
67  HYPRE_Int current_num_elmts; /* current no. of elements stored in stash */
68  HYPRE_Int off_proc_i_indx; /* pointer to first empty space in
69  set_off_proc_i_set */
70  HYPRE_Int *off_proc_i; /* length 2*num_off_procs_elmts, contains info pairs
71  (code, no. of elmts) where code contains global
72  row no. if SetValues, and (-global row no. -1)
73  if AddToValues*/
74  HYPRE_Int *off_proc_j; /* contains column indices */
75  HYPRE_Complex *off_proc_data; /* contains corresponding data */
76  HYPRE_Int cancel_indx; /* number of elements that have to be deleted due
77  to setting values from another processor */
78 } hypre_AuxParCSRMatrix;
