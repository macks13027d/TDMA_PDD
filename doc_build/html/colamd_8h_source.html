161 /* ========================================================================== */
162 /* === Colamd recommended memory size ======================================= */
163 /* ========================================================================== */
164 
165 /*
166  The recommended length Alen of the array A passed to colamd is given by
167  the COLAMD_RECOMMENDED (nnz, n_row, n_col) macro. It returns -1 if any
168  argument is negative. 2*nnz space is required for the row and column
169  indices of the matrix. COLAMD_C (n_col) + COLAMD_R (n_row) space is
170  required for the Col and Row arrays, respectively, which are internal to
171  colamd. An additional n_col space is the minimal amount of "elbow room",
172  and nnz/5 more space is recommended for run time efficiency.
173 
174  This macro is not needed when using symamd.
175 
176  Explicit typecast to int added Sept. 23, 2002, COLAMD version 2.2, to avoid
177  gcc -pedantic warning messages.
178 */
179 
180 #define COLAMD_C(n_col) ((int) (((n_col) + 1) * sizeof (Colamd_Col) / sizeof (int)))
181 #define COLAMD_R(n_row) ((int) (((n_row) + 1) * sizeof (Colamd_Row) / sizeof (int)))
182 
183 #define COLAMD_RECOMMENDED(nnz, n_row, n_col) \
184 ( \
185 ((nnz) < 0 || (n_row) < 0 || (n_col) < 0) \
186 ? \
187  (-1) \
188 : \
189  (2 * (nnz) + COLAMD_C (n_col) + COLAMD_R (n_row) + (n_col) + ((nnz) / 5)) \
190 )
