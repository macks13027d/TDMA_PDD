32 /*--------------------------------------------------------------------------
33  * Parallel CSR Block Matrix
34  *--------------------------------------------------------------------------*/
35 
36 typedef struct
37 {
38  MPI_Comm comm;
39 
40  HYPRE_Int global_num_rows;
41  HYPRE_Int global_num_cols;
42  HYPRE_Int first_row_index;
43  HYPRE_Int first_col_diag;
44 
45  /* need to know entire local range in case row_starts and col_starts
46  are null */
47  HYPRE_Int last_row_index;
48  HYPRE_Int last_col_diag;
49 
50  hypre_CSRBlockMatrix *diag;
51  hypre_CSRBlockMatrix *offd;
52  HYPRE_Int *col_map_offd;
53  /* maps columns of offd to global columns */
54  HYPRE_Int *row_starts;
55  /* array of length num_procs+1, row_starts[i] contains the
56  global number of the first row on proc i,
57  first_row_index = row_starts[my_id],
58  row_starts[num_procs] = global_num_rows */
59  HYPRE_Int *col_starts;
60  /* array of length num_procs+1, col_starts[i] contains the
61  global number of the first column of diag on proc i,
62  first_col_diag = col_starts[my_id],
63  col_starts[num_procs] = global_num_cols */
64 
65  hypre_ParCSRCommPkg *comm_pkg;
66  hypre_ParCSRCommPkg *comm_pkgT;
67 
68  /* Does the ParCSRBlockMatrix create/destroy `diag', `offd', `col_map_offd'? */
69  HYPRE_Int owns_data;
70  /* Does the ParCSRBlockMatrix create/destroy `row_starts', `col_starts'? */
71  HYPRE_Int owns_row_starts;
72  HYPRE_Int owns_col_starts;
73 
74  HYPRE_Int num_nonzeros;
75  HYPRE_Real d_num_nonzeros;
76 
77  /* Buffers used by GetRow to hold row currently being accessed. AJC, 4/99 */
78  HYPRE_Int *rowindices;
79  HYPRE_Complex *rowvalues;
80  HYPRE_Int getrowactive;
81 
82  hypre_IJAssumedPart *assumed_partition; /* only populated if no_global_partition option
83  is used (compile-time option)*/
84 
85 } hypre_ParCSRBlockMatrix;
